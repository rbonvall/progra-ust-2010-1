    program Primos;
    var
        n, d: Integer;
        isPrime: Boolean;
    begin
        Read(n);

        isPrime := true;
        for d := 2 to n - 1 do
        begin
            if n mod d = 0 then
                isPrime := false;
        end;

        if isPrime then
            WriteLn(n, ' es primo')
        else
            WriteLn(n, ' es compuesto')
    end.
